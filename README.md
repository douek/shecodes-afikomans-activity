# &#x202b; צעדים לכתיבת הקוד

&#x202b; בקוד פן המצורף קיבלתן קוד שמגדיר משתנים, הגדרה של פונקציה ועוד קריאה לפונקציה. אנחנו תכף נראה מה צריך להכנס בכל חלק

&#x202b; שימו לב להערות בעברית והעזרו בהן להבין מה הקוד שלנו עושה ואיפה לשים אותו

```javascript
// הגדרה של משתנים
// התבנית של הקוד שאנחנו רוצות
const pattern = [];
// משתנה שיעזור לנו לעקוב אחרי התקדמות בהקלדת הקוד
let current;

// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if () {
    // טיפול במקש נכון
  } else {
    // טיפול עם המקש שנלחץ הוא המקש הלא נכון
  }

  // בדיקה האם כל התבנית הסתיימה
  if () {
    // הצגת הודעה ואיפוס
  }
};

// הקשבה לאירוע לחיצת כפתור והגדרת הפונקציה שתרוץ ברגע שזה יקרה
document.addEventListener();
```
## &#x202b; הגדרת משתנים

&#x202b; המשתנה הראשון הוא בעצם התבנית וזה המקשים שיקושו שיגרמו להתראה

&#x202b; כאן בדוגמה זה קונאמי הקלאסי אבל אתן יכולות לתת איזו תבנית של אותיות שאתן רוצות - **באנגלית**

```javascript
const pattern = ["ArrowUp", "ArrowUp","ArrowDown","ArrowDown","ArrowLeft","ArrowRight","ArrowLeft","ArrowRight","b", "a"];
```

&#x202b; המשתנה השני הוא מספר מעקב שעוזר לנו לעקוב באיזה שלב בתבנית אנחנו נמצאות (בתו הראשון, בתו השני וכו׳)

```javascript
let current = 0;
```

## &#x202b; השלב הבא הוא השלמת הפונקציה שתרוץ בכל פעם שנלחץ מקש

&#x202b; הפונקציה מקבלת פרמטר בשם event

&#x202b; הפרמטר בעצם מכיל פרטים על האירוע שקרה - במקרה הזה לחיצת כפתור ובעזרתו נדע איזה כפתור נלחץ

```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {

};

```

## &#x202b; בדיקת המקש שנלחץ

&#x202b; אנחנו נבדוק האם המקש שנלחץ הוא שווה למקש בתבנית
```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if (event.key === pattern[current]) {
  } 
};

```
&#x202b; ובמידה והוא נכון נקדם את המספר מעקב שלנו בפלוס אחד כדי שבמקש הבא נבדוק אותו מול המקש הבא בתבנית
```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if (event.key === pattern[current]) {
    // טיפול במקש נכון
    current++;
  }
};

```

&#x202b; במידה והמקש הוא לא נכון נחזיר את המעקב ל0
```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if (event.key === pattern[current]) {
    // טיפול במקש נכון
    current++;
  } else {
    // טיפול עם המקש שנלחץ הוא המקש הלא נכון
    current = 0;
  }
};

```

## &#x202b; הצגת התראה ברגע שהצליחו להקיש את הקוד

&#x202b; הצלחה היא כאשר מספר התווים הנכונים שהוקלדו שווים ערך לגודל התבנית
```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if (event.key === pattern[current]) {
    // טיפול במקש נכון
    current++;
  } else {
    // טיפול עם המקש שנלחץ הוא המקש הלא נכון
    current = 0;
  }

  // בדיקה האם כל התבנית הסתיימה
  if (pattern.length === current) {
  }
};

```
&#x202b; במידה והצלחנו נציג התראה למשתמש 

&#x202b; window זה החלון של הדפדפן ושאפשר להפעיל פעולות עליו

&#x202b; אפשר לשים איזה טקסט שאנחנו רוצות בהתראה שקופצת

&#x202b; ונאפס את המעקב כדי לאפשר התחלה מחדש של התבנית 
```javascript
// הפונקציה שתרוץ ברגע שמקש ילחץ
const keyHandler = function (event) {
  // בדיקה האם המקש שנלחץ הוא המקש הבא בתבנית
  if (event.key === pattern[current]) {
    // טיפול במקש נכון
    current++;
  } else {
    // טיפול עם המקש שנלחץ הוא המקש הלא נכון
    current = 0;
  }

  // בדיקה האם כל התבנית הסתיימה
  if (pattern.length === current) {
    // הצגת הודעה ואיפוס
    current = 0;
    window.alert("You found it!");
  }
};

```


&#x202b; והדבר האחרון שנשאר לעשות הוא לרשום את הפונקציה לאירוע של לחיצת מקש


&#x202b;אירועים(Events) אלו פעולות או התרחשויות אשר קורות במערכת שאנחנו מתכנתים בה, והמערכת אומרת לנו  בדרך מסויימת על ההתרחשות שלהם, כך שאנחנו יכולים להגיב להתרחשות של האירוע המסויים בדרך מסויימת שנרצה. לדוגמא אם משתמש לחץ על כפתור באתר, אולי נרצה בתגובה לטפל באירוע הלחיצה הזה על ידי הצגת הודעה כלשהי.

```javascript
document.addEventListener("keydown", keyHandler, false);
```
